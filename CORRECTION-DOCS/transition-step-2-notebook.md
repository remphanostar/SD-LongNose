```python
#@title üì¶ Setup Pinokio (One-Click Solution)
#@markdown Click run to automatically setup Pinokio with tunnel access

import os
import subprocess
import requests
import time
import json
from pathlib import Path

class PinokioCloudSimple:
    def __init__(self):
        self.pinokio_path = "/content/pinokio"
        self.binary_path = None
        self.server_process = None
        self.tunnel_process = None
        self.tunnel_url = None
        
    def setup_directory(self):
        """Create Pinokio directory structure"""
        print("üìÅ Setting up directory...")
        Path(self.pinokio_path).mkdir(parents=True, exist_ok=True)
        os.chdir(self.pinokio_path)
        print(f"‚úÖ Created directory: {self.pinokio_path}")
        
    def download_pinokio(self):
        """Download Pinokio with correct naming"""
        print("üì• Downloading Pinokio...")
        
        # Download with the name the controller expects
        self.binary_path = os.path.join(self.pinokio_path, "Pinokio-linux.AppImage")
        
        if os.path.exists(self.binary_path):
            print("‚úÖ Pinokio already downloaded")
            return True
            
        try:
            # Use the actual download URL but save with correct name
            download_url = "https://github.com/pinokiocomputer/pinokio/releases/download/3.9.0/Pinokio-3.9.0.AppImage"
            
            result = subprocess.run([
                'wget', '-q', '--show-progress', 
                '-O', self.binary_path, 
                download_url
            ], capture_output=True, text=True)
            
            if result.returncode != 0:
                print(f"‚ùå Download failed: {result.stderr}")
                return False
                
            # Make executable
            os.chmod(self.binary_path, 0o755)
            
            # Verify file exists and is executable
            if os.path.exists(self.binary_path) and os.access(self.binary_path, os.X_OK):
                print("‚úÖ Pinokio downloaded and configured")
                return True
            else:
                print("‚ùå Binary not properly configured")
                return False
                
        except Exception as e:
            print(f"‚ùå Download error: {e}")
            return False
    
    def start_pinokio_server(self):
        """Start Pinokio web server (no X11 needed)"""
        print("üöÄ Starting Pinokio web server...")
        
        if not self.binary_path or not os.path.exists(self.binary_path):
            print("‚ùå Pinokio binary not found")
            return False
            
        try:
            # Start in pure web server mode - NO X11 NEEDED!
            env = os.environ.copy()
            env['PINOKIO_APP_PORT'] = '42000'
            
            self.server_process = subprocess.Popen([
                self.binary_path, 
                '--no-sandbox', 
                '--headless'
            ], env=env, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)
            
            # Wait for server to start
            print("‚è≥ Waiting for server startup...")
            for i in range(30):  # 30 second timeout
                try:
                    response = requests.get('http://localhost:42000', timeout=2)
                    if response.status_code == 200:
                        print("‚úÖ Pinokio web server running on port 42000")
                        return True
                except requests.exceptions.RequestException:
                    time.sleep(1)
                    if i % 5 == 0:  # Show progress every 5 seconds
                        print(f"‚è≥ Still waiting... ({i+1}/30)")
                        
            print("‚ùå Server startup timeout - checking process...")
            if self.server_process.poll() is not None:
                stdout, stderr = self.server_process.communicate()
                print(f"‚ùå Process exited with code {self.server_process.returncode}")
                print(f"Error: {stderr[:500]}")  # First 500 chars of error
            return False
            
        except Exception as e:
            print(f"‚ùå Server startup error: {e}")
            return False
    
    def setup_cloudflare_tunnel(self, port=42000):
        """Setup Cloudflare tunnel (no auth required)"""
        print("üåê Setting up Cloudflare tunnel...")
        
        try:
            # Download cloudflared
            cf_path = '/tmp/cloudflared'
            if not os.path.exists(cf_path):
                print("üì• Downloading Cloudflare tunnel...")
                subprocess.run([
                    'wget', '-q', '-O', cf_path,
                    'https://github.com/cloudflare/cloudflared/releases/latest/download/cloudflared-linux-amd64'
                ])
                os.chmod(cf_path, 0o755)
            
            # Start tunnel
            self.tunnel_process = subprocess.Popen([
                cf_path, 'tunnel', '--url', f'http://localhost:{port}'
            ], stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)
            
            # Extract tunnel URL
            print("‚è≥ Starting tunnel...")
            for _ in range(30):
                line = self.tunnel_process.stdout.readline()
                if 'trycloudflare.com' in line:
                    # Extract URL from line
                    for part in line.split():
                        if 'trycloudflare.com' in part:
                            self.tunnel_url = part.strip()
                            print(f"‚úÖ Cloudflare tunnel created: {self.tunnel_url}")
                            return self.tunnel_url
                time.sleep(1)
            
            print("‚ùå Cloudflare tunnel setup failed")
            return None
            
        except Exception as e:
            print(f"‚ùå Cloudflare tunnel error: {e}")
            return None
    
    def setup_localtunnel(self, port=42000):
        """Setup LocalTunnel (no auth required)"""
        print("üåê Setting up LocalTunnel...")
        
        try:
            # Install localtunnel via npm
            print("üì• Installing LocalTunnel...")
            result = subprocess.run(['npm', 'install', '-g', 'localtunnel'], 
                                  capture_output=True, text=True)
            if result.returncode != 0:
                print(f"‚ùå LocalTunnel install failed: {result.stderr}")
                return None
            
            # Start tunnel
            self.tunnel_process = subprocess.Popen([
                'lt', '--port', str(port)
            ], stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)
            
            # Extract URL
            print("‚è≥ Starting LocalTunnel...")
            for _ in range(20):
                line = self.tunnel_process.stdout.readline()
                if 'https://' in line and 'loca.lt' in line:
                    self.tunnel_url = line.strip()
                    print(f"‚úÖ LocalTunnel created: {self.tunnel_url}")
                    return self.tunnel_url
                time.sleep(1)
            
            print("‚ùå LocalTunnel setup failed")
            return None
            
        except Exception as e:
            print(f"‚ùå LocalTunnel error: {e}")
            return None
    
    def setup_tunnel(self):
        """Try multiple tunnel services with fallbacks"""
        print("üåê Setting up public tunnel...")
        
        # Try Cloudflare first (most reliable, no auth)
        url = self.setup_cloudflare_tunnel()
        if url:
            return url
            
        print("‚ö†Ô∏è Cloudflare failed, trying LocalTunnel...")
        url = self.setup_localtunnel()
        if url:
            return url
            
        print("‚ùå All tunnel services failed")
        print("üí° Pinokio is still running locally at: http://localhost:42000")
        return None
    
    def get_status(self):
        """Get current status"""
        status = {
            'directory_ready': os.path.exists(self.pinokio_path),
            'binary_ready': self.binary_path and os.path.exists(self.binary_path),
            'server_running': False,
            'tunnel_active': self.tunnel_url is not None,
            'tunnel_url': self.tunnel_url
        }
        
        # Check if server is responding
        try:
            response = requests.get('http://localhost:42000', timeout=2)
            status['server_running'] = response.status_code == 200
        except:
            status['server_running'] = False
            
        return status
    
    def cleanup(self):
        """Stop all processes"""
        if self.server_process:
            self.server_process.terminate()
        if self.tunnel_process:
            self.tunnel_process.terminate()

# Main execution
try:
    pinokio = PinokioCloudSimple()
    
    # Step-by-step setup
    pinokio.setup_directory()
    
    if not pinokio.download_pinokio():
        raise Exception("Failed to download Pinokio")
    
    if not pinokio.start_pinokio_server():
        raise Exception("Failed to start Pinokio server")
    
    tunnel_url = pinokio.setup_tunnel()
    
    # Final results
    print("\n" + "üéâ SETUP COMPLETE! " + "üéâ")
    print("=" * 50)
    
    if tunnel_url:
        print(f"üåç Public URL: {tunnel_url}")
        print("üì± Click the link above to access Pinokio!")
    else:
        print("üñ•Ô∏è  Local URL: http://localhost:42000")
        print("üí° Access locally or setup port forwarding")
    
    print("\nüìã What you can do now:")
    print("‚Ä¢ Browse the 'Discover' tab for 500+ AI tools")
    print("‚Ä¢ Install Stable Diffusion, ComfyUI, etc. with one click")
    print("‚Ä¢ Access all tools through the web interface")
    
    # Save status for later cells
    _pinokio_instance = pinokio
    _pinokio_url = tunnel_url or "http://localhost:42000"
    
except Exception as e:
    print(f"\n‚ùå Setup failed: {e}")
    print("\nüîß Troubleshooting:")
    print("1. Check if GPU runtime is enabled")
    print("2. Restart the notebook and try again")
    print("3. Check the status cell below for details")
```
